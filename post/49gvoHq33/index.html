简单的代码生成程序

发布于 2019-11-28 | 分类于 编译原理题解报告 | 3分钟 | 446字数

通过三地址代码序列生成计算机的目标代码,在生成算法中,对寄存器的使用顺序为:寄存器中存有 > 空寄存器 > 内存中存有 > 以后不再使用 > 最远距离使用

输入格式:
给定输出的三地址代码的个数和寄存器的个数.所有的变量为大写字母,寄存器的数量不超过9

输出格式:
参照示例格式输出,不需要将最后的寄存器中的值写回内存
不再使用变量不用写回内存

输入样例:
4 2
T:=A-B
U:=A-C
V:=T+U
W:=V+U
输出样例:
LD R0, A
SUB R0, B
LD R1, A
SUB R1, C
ADD R0, R1
ADD R0, R1

#include<bits/stdc++.h>
using namespace std;
int n,m,nr=0;
string s[100];
char r[15];
int inr(char c){
    for(int i=0;i<m;i++)if(r[i]==c) return i;
    return -1;
}
int gul(int pos,char c){
    for(int i=pos;i<=n;i++)if(s[i][3]==c||s[i][5]==c)return i;
    return n+1;
}
int getr(int pos){
    if(nr<m) return nr++;
    int ans=-1,maxx=-1;
    for(int i=0;i<m;i++){
        int ne= gul(pos,r[i]);
        if(ne>maxx) maxx=ne,ans=i;
    }
    return ans;
}
void printop(char c){
    if(c=='+') printf("ADD ");
    if(c=='*') printf("MUL ");
    if(c=='-') printf("SUB ");
}
void printr(char c){
    int pos= inr(c);
    if(pos==-1)printf("%c\n",c);
    else printf("R%d\n",pos);
}
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>s[i];
    for(int i=1;i<=n;i++){
        int pos = inr(s[i][3]);
        if(pos==-1){
            pos=getr(i);
            if(r[pos]&&gul(i,r[pos])<n+1)
                printf("ST R%d, %c\n",pos,r[pos]);
            printf("LD R%d, %c\n",pos,s[i][3]);
            r[pos]=s[i][3];
        }
        printop(s[i][4]);
        printf("R%d, ",pos);
        printr(s[i][5]);
        r[pos]=s[i][0];
    }
}
